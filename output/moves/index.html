library(animation)
library(png)
while (t <= N) {
    if (t < 10) 
        tt <- paste("000", t, sep = "")
    else if (t < 100) 
        tt <- paste("00", t, sep = "")
    else if (t < 1000) 
        tt <- paste("0", t, sep = "")
    else tt <- t
    img <- readPNG(paste(name, tt, ".png", sep = ""), native = T, 
        info = T)
    plot(0, 0, xlim = c(0, 20), ylim = c(0, 20), xaxt = "n", 
        yaxt = "n", xlab = "", ylab = "", t = "n")
    rasterImage(img, -1, -1, 21, 21)
    t <- t + step
}
